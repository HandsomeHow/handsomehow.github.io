题意：给定一棵N个点树,q次询问,每次询问以某个结点为根的子树的重心。(N<=300000,q<=300000)

题解：

因为询问次数巨大, 所以我们要预处理所有结点的答案。

因为重心一定是在重链上的,所以我们只要对树先DFS一次找到重链，然后暴力的去找重心就可以了。

const int maxn = 333333;
int ans[maxn], sz[maxn], son[maxn];
int n, q;
int par[maxn];

vector<int>E[maxn];

void dfs(int u, int p){
	ans[u] = u;
	if(E[u].size() == 0)
		son[u] = u;
	sz[u] = 1;
	for(auto it:E[u]){
		if(it == p)
			continue;
		dfs(it,u);
		sz[u] += sz[it];
		if(sz[it] > sz[son[u]])
			son[u] = it;
	}
	ans[u] = ans[son[u]];
	//printf("---%d %d %d\n",u,son[u],ans[u]);
	while(sz[u] - sz[ans[u]] > sz[u]/2 && ans[u] != u){
		ans[u] = par[ans[u]];
	}
}
