#题意:
给你一个S串和一个T串(长度<=1e5),让你在T串种删去一个长度尽可能少的子串(长度可为0,可为T的长度),使得删除后的T串是S的一个子序列.
#题解:
直接考虑删去可能有点困难,于是我们反过来想,从T串左边选一部分,右边选一部分,然后使得他成为S串的子串.

于是我们可以想到一个two point维护两个串的下标,从左往右遍历,维护s串的每一个字符能匹配到T串的哪个位置,然后再反向遍历一遍,最后再遍历一下整个串就可以得到答案.

```c++
/*************************************************************************

#include <bits/stdc++.h>
using namespace std;

#define rep(i,a,b) for(int i=(a);i<=(b);i++)
#define per(i,a,b) for(int i=(a);i>=(b);i--)
#define pb push_back
#define mp make_pair
#define cl(a) memset((a),0,sizeof(a))
const int maxn = 1e5+5;
int idl[maxn], idr[maxn];
char s[maxn], t[maxn];
int main(){
	cl(idl);
	scanf("%s%s",s+1,t+1);
	int len1 = strlen(s+1);
	int len2 = strlen(t+1);
	rep(i,0,len1+1)
		idr[i] = len2+1;
	int ids = 0, idt = 0;
	while(idt + 1 <= len2 && ids +1 <= len1){
		ids++;
		if(t[idt+1] == s[ids]){
			idt++;
			idl[ids] = idt;
		}
	}
	ids = len1 + 1, idt = len2 + 1;
	while(idt - 1 > 0 && ids - 1 > 0){
		ids--;
		if(t[idt-1] == s[ids]){
			idt--;
			idr[ids] = idt;
		}
	}
	rep(i,1,len1)
		if(idl[i-1] > idl[i])
			idl[i] = idl[i-1];
	per(i,len1,1)
		if(idr[i+1] < idr[i])
			idr[i] = idr[i+1];

	int ansidx = -1, anslen = -1;
	rep(i,0,len1){
		if(idl[i] >= idr[i+1]){
			ansidx = i;
			anslen = len1;
		}else{
			int tot = idl[i] + len2 - idr[i+1] + 1;
			if(tot > anslen){
				anslen = tot;
				ansidx = i;
			}
		}
	}
	if(anslen <= 0)
		puts("-");
	else if(anslen >= len2){
		printf("%s\n",t+1);
	}else{
		rep(i,1,idl[ansidx])
			cout<<t[i];
		rep(i,idr[ansidx+1],len2)
			cout<<t[i];
		cout<<endl;
	}

	return 0;	
}
```
